# 逻辑文件的组织

  • js:
    • components: Vue 组件
    • features: 各种功能模块
    • app.js: 加载各种功能
    • feature.js: 注册所有的功能模块
    • loader.js: 功能模块的加载器
    • vue.js: 处理 Vue 相关功能

修改 resources/js/app.js:

import 'jquery';
import 'popper.js';
import 'bootstrap';

import './feature';
import './vue';

添加 resources/js/vue.js:

import Vue from 'vue';
import ExampleComponent from './components/ExampleComponent.vue';

new Vue({
  el: '#app',
  components: {
    ExampleComponent
  }
});

# 实现功能模块

创建 resources/js/loader.js:

window.Feature = window.Feature || {};
window.RegisteredFeatures = window.RegisteredFeatures || [];

window.onload = function() {
  window.RegisteredFeatures.forEach(feature => feature());
};

修改 webpack.mix.js:

mix
  .js('resources/js/loader.js', 'public/js')
  .js('resources/js/app.js', 'public/js')
  .sass('resources/sass/app.scss', 'public/css');

在 resources/views/widgets/_head.blade.php 中优先注册 loader.js。原因是 app.js 是一个 defer 模块，而我们需要保证页面下方的 RegisteredFeatures 和 Features 两个全局变量在 app.js 执行前就被定义。

<head>
  <!-- Scripts -->
  <script src="{{ asset('js/loader.js') }}"></script>
  <script src="{{ asset('js/app.js') }}" defer></script>
</head>

创建 resources/js/feature.js:

import { SayHi } from './features/SayHi';

window.Feature.SayHi = (...args) => new SayHi(...args);

创建 resources/js/features/SayHi.js:

export class SayHi {
  constructor(greeting = 'hello, world!') {
    console.log(greeting);
  }
}

编辑 resources/views/layouts/app.blade.php，在 <body> 标签底部添加 scripts 区块:

<body>
<div id="app" class="layouts-app">
  ...
</div>

@yield('scripts')
</body>

在 resources/views/index/index.blade.php 中使用 SayHi 功能:

@section('scripts')
  <script>
  RegisteredFeatures.push(function () {
    Feature.SayHi('hello, world')
  });
  </script>
@endsection

# 使用 Typescript

注意: 本人对 TS 的了解并不算特别深，尤其是在类型声明这部分，几乎相当于 AnyScript 了。

重命名所有 .js 文件为 .ts。

更新 webpack.mix.js:

mix
  .ts('resources/js/loader.ts', 'public/js')
  .ts('resources/js/app.ts', 'public/js')
  .sass('resources/sass/app.scss', 'public/css');

执行 npm run watch，Laravel-Mix 将自动安装一些 TS 相关依赖，如果出现报错也没关系，先停掉 npm 进程。

安装 jQuery 的类型生命文件和 Vue 的 TS 支持:

npm i -D @types/jquery vue-class-component

在项目根目录下创建 tsconfig.json:

{
    "compilerOptions": {
        "module": "commonjs",
        "target": "es5",
        "noImplicitAny": false,
        "sourceMap": true,
        "experimentalDecorators": true
    },
    "include": [
        "resources/js/**/*"
    ]
}

调整 loader.js:

interface IFeature {
  [propName: string]: () => void;
}

interface Window {
  Feature: IFeature;
  RegisteredFeatures: Function[];
}

window.Feature = window.Feature || {};
window.RegisteredFeatures = window.RegisteredFeatures || [];

window.onload = function() {
  window.RegisteredFeatures.forEach(feature => feature());
};

调整 app.ts:

import * as $ from 'jquery';
import * as Popper from 'popper.js';
import 'bootstrap';

(window as any).$ = $;
(window as any).Popper = Popper;

import './feature';
import './vue';

调整 feature.ts:

import { SayHi } from './features/SayHi';

const WF = (window as any).Feature;

WF.SayHi = (...args) => new SayHi(...args);

调整 js/components/ExampleComponent.vue:

<template>
  <div class="container">
    <div class="row justify-content-center">
      <div class="col-md-8">
        <div class="card">
          <div class="card-header">Example Component</div>

          <div class="card-body">
            I'm an example component.
          </div>
        </div>
      </div>
    </div>
  </div>
</template>

<script lang="ts">
  import Vue from 'vue';
  import Component from 'vue-class-component';

  @Component
  export default class ExampleComponent extends Vue {
    private mounted() {
      console.log('Component mounted.');
    }
  }
</script>

重新执行 npm run watch，检查是否存在报错。也可以在页面上使用 example-component 组件检验 Vue 组件的 TS 支持情况。
