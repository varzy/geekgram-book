# 初始化项目

# 创建项目

使用 composer 创建一个全新的 Laravel 项目:

composer create-project --prefer-dist laravel/laravel geekgram

# 初始化 Git 仓库

首先在 .gitignore 中添加一些必要的忽略:

.idea
.php_cs.cache

执行以下命令:

cd geekgram

git init
git add .
git commit -m "init"

# 使用 php-cs-fixer 自动格式化 PHP 代码

通过 composer 安装 php-cs-fixer:

composer require --dev friendsofphp/php-cs-fixer

在项目根目录添加 .php_cs.dist:

<?php

$finder = PhpCsFixer\Finder::create()
    ->exclude('node_modules')
    ->exclude('vendor')
    ->exclude('storage')
    ->in(__DIR__)
;

return PhpCsFixer\Config::create()
    ->setRules([
        '@Symfony' => true,
        'binary_operator_spaces' => [
            // 对齐双箭头操作符
            'align_double_arrow' => true,
            // 不对齐赋值操作符
            'align_equals' => false,
        ],
    ])
    ->setFinder($finder)
;

在 composer.json 中添加执行脚本:

"scripts": {
  "lint": "php-cs-fixer fix --diff --dry-run",
  "fix": "php-cs-fixer fix"
}
